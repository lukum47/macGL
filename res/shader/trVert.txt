#version 460 
layout(location = 0) in vec3 vertex_position;
layout(location = 1) in vec3 Normal;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;
out vec3 norm;
out vec3 fragPos;

void main() {
	norm = Normal;
	fragPos = vec3(model * vec4(vertex_position, 1.0));
	gl_Position = projection * view * model * vec4(vertex_position, 1.0);

}
